81 def to_geo_data(
82 self,
83 profile_data: ProfileData,
84 merge_strategy: Optional[MergeStrategy] = None,
85 ) -> GeoData:
86 """!
87 @brief Convert ProfileData to GeoData for interpolation
88
89 @brief Handles both 6+6 and 3+3 modes with merge strategies.
90
91 @param profile_data Profile data to convert
92 @param merge_strategy Optional merge strategy (overrides instance strategy)
93 @return GeoData ready for interpolation
94 @exception ProfileModeError If conversion fails
95 """
96 # Auto-detect movement mode if enabled
97 if self._auto_detect:
98 self.auto_detect_movement(profile_data)
99
100 # Use provided strategy or instance strategy
101 strategy = merge_strategy or self._merge_strategy
102
103 # If no strategy specified for 3+3 mode, use mean
104 if profile_data.movement_mode == "3+3" and strategy is None:
105 strategy = MergeStrategyFactory.create("mean")
106
107 # Perform conversion
108 if profile_data.movement_mode == "6+6":
109 return self._convert_6_6_mode(profile_data)
110 else: # "3+3"
111 if strategy is None:
112 strategy = MergeStrategyFactory.create("mean")
113 return self._convert_3_3_mode(profile_data, strategy)

115 def _convert_6_6_mode(self, profile_data: ProfileData) -> GeoData:
116 """!
117 @brief Convert ProfileData in 6+6 mode
118
119 @brief All 6 sensors move together, take all readings.
120
121 @param profile_data Profile data in 6+6 mode
122 @return GeoData with all sensor readings
123 """
124 # Collect all readings
125 all_readings: list[SensorReading] = []
126 for step in profile_data.steps:
127 all_readings.extend(step.readings)
128
129 # Extract positions and values
130 points = np.array([[r.x, r.y, r.z] for r in all_readings], dtype=np.float64)
131 values = np.array([r.value for r in all_readings], dtype=np.float64)
132
133 return GeoData.from_arrays(
134 points=points,
135 values=values,
136 metadata={
137 "movement_mode": "6+6",
138 "n_steps": str(profile_data.n_steps),
139 "source": "profile_data",
140 },
141 )

189 def validate_profile_data(self, profile_data: ProfileData) -> bool:
190 """!
191 @brief Validate profile data integrity
192
193 @brief Checks that profile data is consistent and ready for conversion.
194
195 @param profile_data Profile data to validate
196 @return True if valid
197 @exception ProfileModeError If validation fails
198 """
199 # Check that profile has steps
200 if profile_data.n_steps == 0:
201 raise ProfileModeError("Profile data has no steps")
202
203 # Check that each step has exactly 6 readings
204 for step in profile_data.steps:
205 if step.n_readings != 6:
206 raise ProfileModeError(
207 f"Step {step.step_number} has {step.n_readings} readings, expected 6",
208 step=step.step_number,
209 )
210
211 # Validate movement mode
212 if profile_data.movement_mode not in ("6+6", "3+3"):
213 raise ProfileModeError(f"Invalid movement mode: {profile_data.movement_mode}")
214
215 return True

218def merge_overlapping_measurements(
219 readings: list[SensorReading],
220 strategy: MergeStrategy,
221) -> tuple[NDArray[np.float64], NDArray[np.float64]]:
222 """!
223 @brief Merge overlapping measurements at same position
224
225 @brief Groups readings by (x, y, z) position and applies merge strategy.
226
227 @param readings List of sensor readings (may have overlaps)
228 @param strategy Merge strategy to apply
229 @return Tuple of (points_array, values_array)
230 - points_array: (N, 3) array of unique positions
231 - values_array: (N,) array of merged values
232
233 @example
234 @code
235 >>> from src.profile.merging import MeanMergeStrategy
236 >>> from src.profile.integration import merge_overlapping_measurements
237 >>>
238 >>> # Two sensors measured same position
239 >>> readings = [
240 ... SensorReading(sensor_id=1, step=1, x=0, y=0, z=0, value=10.0),
241 ... SensorReading(sensor_id=4, step=1, x=0, y=0, z=0, value=12.0),
242 ... ]
243 >>>
244 >>> strategy = MeanMergeStrategy()
245 >>> points, values = merge_overlapping_measurements(readings, strategy)
246 >>> # points: [[0, 0, 0]]
247 >>> # values: [11.0] # Average of 10 and 12
248 @endcode
249 """
250 # Group readings by position
251 position_groups: dict[tuple[float, float, float], list[float]] = {}
252
253 for reading in readings:
254 pos_key = (reading.x, reading.y, reading.z)
255 if pos_key not in position_groups:
256 position_groups[pos_key] = []
257 position_groups[pos_key].append(reading.value)
258
259 # Merge each group
260 merged_points: list[list[float]] = []
261 merged_values: list[float] = []
262
263 for pos, vals in position_groups.items():
264 merged_points.append([pos[0], pos[1], pos[2]])
265
266 # Apply merge strategy
267 vals_array = np.array(vals, dtype=np.float64)
268 merged_value = strategy.merge(vals_array)
269 merged_values.append(merged_value)
270
271 # Convert to numpy arrays
272 points_array = np.array(merged_points, dtype=np.float64)
273 values_array = np.array(merged_values, dtype=np.float64)
274
275 return points_array, values_array
