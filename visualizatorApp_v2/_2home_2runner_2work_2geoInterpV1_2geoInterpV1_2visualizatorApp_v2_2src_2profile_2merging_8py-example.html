48class MeanMergeStrategy(MergeStrategy):
49 """! @brief Merge using arithmetic mean
50
51 @brief Calculates the average of all values.
52 This is the default and most commonly used strategy.
53
54 @example
55 @code
56 >>> strategy = MeanMergeStrategy()

58 >>> merged = strategy.merge(values) # Returns 11.0
59 @endcode

62 def merge(self, values: NDArray[np.float64]) -> float:
63 """!
64 @brief Merge using arithmetic mean

69 """
70 if len(values) == 0:
71 raise ProfileModeError("Cannot merge empty values array")

120class WeightedMergeStrategy(MergeStrategy):
121 """! @brief Merge using weighted average
122
123 @brief Calculates weighted average with custom weights.
124 Useful when some measurements are more reliable.
125
126 @example
127 @code
128 >>> strategy = WeightedMergeStrategy(weights=np.array([0.5, 0.3, 0.2]))
129 >>> values = np.array([10.0, 12.0, 11.0])
130 >>> merged = strategy.merge(values) # Returns 10.8
131 @endcode
132 """
133
134 def __init__(self, weights: NDArray[np.float64] | None = None) -> None:
135 """!
136 @brief Constructor
137
138 @param weights Optional weights for each value.
139 If None, uses equal weights (same as mean).
140 Must have same length as values being merged.
141 """
142 self._weights = weights
143
144 def merge(self, values: NDArray[np.float64]) -> float:
145 """!
146 @brief Merge using weighted average
147
148 @param values Array of values to merge
149 @return Weighted average of values
150 @exception ProfileModeError If values array is empty or weights mismatch
151 """
152 if len(values) == 0:
153 raise ProfileModeError("Cannot merge empty values array")
154
155 if self._weights is None:
156 # Equal weights -> same as mean
157 return float(np.mean(values))
158
159 if len(self._weights) != len(values):
160 raise ProfileModeError(
161 f"Weights length ({len(self._weights)}) must match values length ({len(values)})"
162 )
163
164 # Normalize weights to sum to 1
165 normalized_weights = self._weights / np.sum(self._weights)
166
167 return float(np.sum(values * normalized_weights))
168
169 def get_name(self) -> str:
170 """!
171 @brief Get strategy name
172
173 @return "weighted"
174 """
175 return "weighted"
176
177 def set_weights(self, weights: NDArray[np.float64]) -> None:
178 """!
179 @brief Update weights for merging
180
181 @param weights New weights array
182 """
183 self._weights = weights

204 def __init__(self, merge_func: Callable[[NDArray[np.float64]], float]) -> None:
205 """!
206 @brief Constructor
207

209 """
210 self._merge_func = merge_func

238class MergeStrategyFactory:
239 """! @brief Factory for creating merge strategies
240
241 @brief Provides convenient methods to create merge strategies
242 from string names or parameters.
243 """
244
245 _strategies: dict[str, type[MergeStrategy]] = {
246 "mean": MeanMergeStrategy,
247 "median": MedianMergeStrategy,
248 "weighted": WeightedMergeStrategy,
249 "custom": CustomMergeStrategy,
250 }
251
252 @classmethod
253 def create(cls, strategy_name: str, **kwargs: object) -> MergeStrategy:
254 """!
255 @brief Create merge strategy by name
256
257 @param strategy_name Name of strategy ("mean", "median", "weighted", "custom")
258 @param kwargs Additional arguments for strategy constructor
259 @return Merge strategy instance
260 @exception ProfileModeError If strategy name is unknown
261
262 @example
263 @code
264 >>> # Create mean strategy
265 >>> strategy = MergeStrategyFactory.create("mean")
266 >>>
267 >>> # Create weighted strategy with custom weights
268 >>> strategy = MergeStrategyFactory.create("weighted",
269 ... weights=np.array([0.5, 0.5]))
270 >>>
271 >>> # Create custom strategy
272 >>> strategy = MergeStrategyFactory.create("custom",
273 ... merge_func=lambda v: np.max(v))
274 @endcode
275 """
276 strategy_name_lower = strategy_name.lower()
277
278 if strategy_name_lower not in cls._strategies:
279 available = ", ".join(cls._strategies.keys())
280 raise ProfileModeError(
281 f"Unknown merge strategy: '{strategy_name}'. Available: {available}"
282 )
283
284 strategy_class = cls._strategies[strategy_name_lower]
285 return strategy_class(**kwargs)
286
287 @classmethod
288 def register_strategy(cls, name: str, strategy_class: type[MergeStrategy]) -> None:
289 """!
290 @brief Register a custom merge strategy
291
292 @param name Name for the strategy
293 @param strategy_class Strategy class (must inherit from MergeStrategy)
294 """
295 if not issubclass(strategy_class, MergeStrategy):
296 raise ProfileModeError(
297 f"Strategy class must inherit from MergeStrategy: {strategy_class}"
298 )
299
300 cls._strategies[name.lower()] = strategy_class
301
302 @classmethod
303 def available_strategies(cls) -> list[str]:
304 """!
305 @brief Get list of available strategy names
306
307 @return List of strategy names
308 """
309 return list(cls._strategies.keys())
