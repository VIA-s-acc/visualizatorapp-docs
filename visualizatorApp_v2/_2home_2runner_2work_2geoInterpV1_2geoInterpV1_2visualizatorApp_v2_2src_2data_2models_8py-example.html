311 """!
312 @brief Add a step to the profile data
313
314 @param step SensorStep to add

344 def to_geo_data(self) -> GeoData:
345 """!
346 @brief Convert to GeoData for interpolation
347
348 @brief Converts sensor readings to GeoData format.
349 Merge strategy is applied to overlapping measurements.
350
351 @return GeoData instance
352 """
353 # Collect all readings
354 all_readings = []
355 for step in self.steps:
356 all_readings.extend(step.readings)
357
358 # Apply merge strategy based on movement mode
359 if self.movement_mode == "6+6":
360 # All 6 sensors move together - take all readings
361 points = np.array([[r.x, r.y, r.z] for r in all_readings])
362 values = np.array([r.value for r in all_readings])
363 else: # "3+3"
364 # Sensors 1-3 and 4-6 may have overlapping measurements
365 # Apply merge strategy
366 points = []
367 values = []
368
369 # Group by position (x, y, z)
370 position_groups: dict[tuple[float, float, float], list[float]] = {}
371
372 for reading in all_readings:
373 pos_key = (reading.x, reading.y, reading.z)
374 if pos_key not in position_groups:
375 position_groups[pos_key] = []
376 position_groups[pos_key].append(reading.value)
377
378 # Apply merge strategy
379 for pos, vals in position_groups.items():
380 points.append([pos[0], pos[1], pos[2]])
381
382 if self.merge_strategy == "mean":
383 values.append(np.mean(vals))
384 elif self.merge_strategy == "median":
385 values.append(np.median(vals))
386 else: # Default to mean
387 values.append(np.mean(vals))
388
389 points = np.array(points)
390 values = np.array(values)
391
392 return GeoData(
393 x=points[:, 0],
394 y=points[:, 1],
395 z=points[:, 2],
396 values=values,
397 metadata={
398 "movement_mode": self.movement_mode,
399 "merge_strategy": self.merge_strategy,
400 "n_steps": str(self.n_steps),
401 "source": "profile_data",
402 },
403 )
404
405 def to_dict(self) -> dict[str, Any]:
406 """!
407 @brief Convert to dictionary
408
409 @return Dictionary representation
410 """
411 return {
412 "steps": [s.to_dict() for s in self.steps],
413 "movement_mode": self.movement_mode,
414 "merge_strategy": self.merge_strategy,
415 "n_steps": self.n_steps,
416 }

423 f"merge_strategy='{self.merge_strategy}')"
