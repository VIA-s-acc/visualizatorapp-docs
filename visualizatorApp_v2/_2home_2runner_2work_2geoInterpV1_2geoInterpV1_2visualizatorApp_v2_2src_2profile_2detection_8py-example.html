102 def detect(self, profile_data: ProfileData) -> DetectionResult:
103 """!
104 @brief Detect movement mode from profile data
105
106 @brief Analyzes sensor positions across all steps to determine movement mode.
107
108 @param profile_data Profile data with sensor readings
109 @return Detection result with mode and confidence
110 @exception ProfileModeError If insufficient data for detection
111 """
112 if profile_data.n_steps == 0:
113 raise ProfileModeError("Cannot detect movement mode: no steps in profile data")
114
115 # For single step, default to 6+6
116 if profile_data.n_steps == 1:
117 return DetectionResult(
118 mode=MovementMode.MODE_6_6,
119 confidence=1.0,
120 group1_sensor_ids=[1, 2, 3, 4, 5, 6],
121 group2_sensor_ids=None,
122 reasoning="Single step detected - assuming 6+6 mode",
123 )
124
125 # Analyze position patterns across steps
126 return self._analyze_movement_patterns(profile_data)

128 def _analyze_movement_patterns(self, profile_data: ProfileData) -> DetectionResult:
129 """!
130 @brief Analyze movement patterns across steps
131
132 @brief Compares sensor positions to detect grouping patterns.
133
134 @param profile_data Profile data to analyze
135 @return Detection result
136 """
137 # Extract sensor positions for each step
138 step_positions = self._extract_step_positions(profile_data)
139
140 # Check for 3+3 pattern: sensors 1-3 move together, 4-6 move together
141 result_3_3 = self._check_3_3_pattern(step_positions)
142
143 if result_3_3.confidence >= self._min_confidence:
144 return result_3_3
145
146 # Default to 6+6
147 return DetectionResult(
148 mode=MovementMode.MODE_6_6,
149 confidence=1.0 - result_3_3.confidence, # Inverse of 3+3 confidence
150 group1_sensor_ids=[1, 2, 3, 4, 5, 6],
151 group2_sensor_ids=None,
152 reasoning="All sensors appear to move together as a single group",
153 )

155 def _extract_step_positions(
156 self, profile_data: ProfileData
157 ) -> dict[int, dict[int, NDArray[np.float64]]]:
158 """!
159 @brief Extract sensor positions for each step
160
161 @param profile_data Profile data
162 @return Dictionary mapping step_number -> sensor_id -> position array [x, y, z]
163 """
164 positions: dict[int, dict[int, NDArray[np.float64]]] = {}
165
166 for step in profile_data.steps:
167 step_positions: dict[int, NDArray[np.float64]] = {}
168 for reading in step.readings:
169 step_positions[reading.sensor_id] = np.array(
170 [reading.x, reading.y, reading.z], dtype=np.float64
171 )
172 positions[step.step_number] = step_positions
173
174 return positions

176 def _check_3_3_pattern(
177 self, step_positions: dict[int, dict[int, NDArray[np.float64]]]
178 ) -> DetectionResult:
179 """!
180 @brief Check if data matches 3+3 pattern
181
182 @brief In 3+3 mode:
183 - Sensors 1-3 should have the same positions across steps
184 - Sensors 4-6 should have the same positions across steps
185 - Groups 1-3 and 4-6 may have different positions
186
187 @param step_positions Dictionary of positions per step
188 @return Detection result for 3+3 pattern
189 """
190 # Count how many steps show 3+3 pattern
191 matching_steps = 0
192 total_steps = len(step_positions)
193
194 for step_num, positions in step_positions.items():
195 if self._is_3_3_step(positions):
196 matching_steps += 1
197
198 # Calculate confidence based on proportion of matching steps
199 confidence = matching_steps / total_steps if total_steps > 0 else 0.0
200
201 # Only report 3+3 if at least some steps match
202 if confidence > 0.3:
203 return DetectionResult(
204 mode=MovementMode.MODE_3_3,
205 confidence=confidence,
206 group1_sensor_ids=[1, 2, 3],
207 group2_sensor_ids=[4, 5, 6],
208 reasoning=f"3+3 pattern detected in {matching_steps}/{total_steps} steps",
209 )
210
211 # Low confidence 3+3 result (will default to 6+6)
212 return DetectionResult(
213 mode=MovementMode.MODE_3_3,
214 confidence=0.0,
215 group1_sensor_ids=[1, 2, 3],
216 group2_sensor_ids=[4, 5, 6],
217 reasoning="No clear 3+3 pattern detected",
218 )

220 def _is_3_3_step(self, positions: dict[int, NDArray[np.float64]]) -> bool:
221 """!
222 @brief Check if a single step matches 3+3 pattern
223
224 @param positions Dictionary mapping sensor_id -> position
225 @return True if sensors 1-3 have same positions and 4-6 have same positions
226 """
227 # Get positions for each group
228 group1_positions = [positions[i] for i in [1, 2, 3]]
229 group2_positions = [positions[i] for i in [4, 5, 6]]
230
231 # Check if all sensors in group 1 have the same position
232 group1_same = self._positions_are_equal(group1_positions)
233
234 # Check if all sensors in group 2 have the same position
235 group2_same = self._positions_are_equal(group2_positions)
236
237 # Check if groups have different positions (optional, for stronger pattern)
238 # For basic 3+3 detection, just need groups to be internally consistent
239 return group1_same and group2_same

241 def _positions_are_equal(self, positions: list[NDArray[np.float64]]) -> bool:
242 """!
243 @brief Check if all positions are approximately equal
244
245 @param positions List of position arrays
246 @return True if all positions are within distance threshold
247 """
248 if len(positions) < 2:
249 return True
250
251 reference = positions[0]
252 for pos in positions[1:]:
253 distance = np.linalg.norm(reference - pos)
254 if distance > self._distance_threshold:
255 return False
256
257 return True
